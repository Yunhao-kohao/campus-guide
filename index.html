默认为英雄 target.health -= self.damage return f"对英雄造成 {self.damage} 点伤害！" else: return "请指定目标。" # ======================== 游戏实体 ======================== class Minion: """战场上的随从实例""" def __init__(self, name: str, attack: int, health: int, cost: int): self.name = name self.attack = attack self.health = health self.cost = cost # 原始费用，暂未使用 self.attacks_this_turn = False # 召唤当回合不能攻击 def can_attack(self): return not self.attacks_this_turn and self.attack > 0 def take_damage(self, damage: int): self.health -= damage def __repr__(self): return f"{self.name}({self.attack}/{self.health}){'<无法攻击>' if not self.can_attack() else ''}" class Hero: """英雄类""" def __init__(self, name: str, health: int = 30, deck: List[Card] = None): self.name = name self.health = health self.max_mana = 0 self.current_mana = 0 self.deck = deck if deck else [] # 牌库（卡牌对象） self.hand = [] # 手牌 self.board = [] # 战场随从 self.armor = 0 # 护甲（可扩展） def draw_cards(self, num: int): for _ in range(num): if self.deck: card = self.deck.pop(0) # 从牌库顶抽牌 self.hand.append(card) if len(self.hand) > 10: # 攻击敌方随从 enemy_minion = target.board[target_minion_index] # 交换伤害 enemy_minion.health -= minion.attack minion.health -= enemy_minion.attack res = f"{minion.name} 攻击 {enemy_minion.name}！" # 检查死亡 if enemy_minion.health <= 0: target.board.pop(target_minion_index) res += f" {enemy_minion.name} 被消灭。" if minion.health <= 0: attacker.board.pop(minion_index) res += f" {minion.name} 被消灭。" minion.attacks_this_turn = True return res elif target_minion_index == -1: # 攻击敌方英雄 target.health -= minion.attack minion.attacks_this_turn = True return f"{minion.name} 攻击英雄，造成 {minion.attack} 点伤害！" else: return "目标无效" def ai_turn(self): """简易AI逻辑""" ai_h = self.ai opp = self.player print(f"\n[电脑回合] 法力:{ai_h.current_mana}/{ai_h.max_mana}") #
